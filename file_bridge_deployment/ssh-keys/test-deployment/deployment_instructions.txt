   chmod 600 ~/.ssh/authorized_keys
   chmod 700 ~/.ssh
   rm /tmp/id_rsa.pub

3. Test the connection:
   ssh -i /home/keldamar/repos/starbridge-platform/file_bridge_deployment/ssh-keys/test-deployment/id_rsa user@target-host

4. Update your Makefile deployment with:
   HOST=target-host
   USER=username
